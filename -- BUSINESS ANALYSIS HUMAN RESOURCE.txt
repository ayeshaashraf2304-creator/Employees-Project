-- BUSINESS ANALYSIS: EMPLOYEES DATA ANALYSIS

-- Question 1: What is the total employee count and average salary for each departnment?

SELECT departnment, COUNT(*) AS employee_count, AVG(REPLACE(salary, '$', ''):: numeric) AS average_salary
FROM employees
GROUP BY departnment
ORDER BY average_salary DESC;


-- Question 2: Is there a salary gap between genders across the company?

SELECT gender, COUNT(*) AS employee_count,
AVG (REPLACE(salary, '$', ''):: numeric) as average_salary
FROM employees
GROUP BY gender;

-- Question 3: Find the top 3 highest-paid employees in the Engineering departnment?

SELECT first_name, last_name, salary
FROM employees
WHERE departnment = 'Engineering'
ORDER BY salary DESC
LIMIT 3;


-- Question 4: How many employees were hired each year?

SELECT extract(YEAR from "hire_date" )AS hire_year,COUNT(*) AS employee_count
FROM employees
GROUP BY hire_year
ORDER BY hire_year ASC;
